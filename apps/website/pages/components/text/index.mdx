export function BlogIndex() {
  const currentPageTitle = "Text Components";
  return getPagesUnderRoute('/components/text')
    .filter(page => page.meta?.title !== currentPageTitle && page.frontMatter?.title !== currentPageTitle)
    .map(page => {
      const fullUrl = page.frontMatter?.image;
      const srcPath = fullUrl?.replace(/^https?:\/\/[^/]+/, '');
      return (
        <ShowcaseCard title={page.meta?.title || page.frontMatter?.title || page.name} href={page.route}>
          <Image src={srcPath} alt={page.frontMatter?.title} width={500} height={300} />
        </ShowcaseCard>
      );
    });
}

export const ShowcaseCard = Object.assign(
  Cards.Card.bind(),
  {
    defaultProps: {
      image: true,
      arrow: true,
    },
    displayName: 'ShowcaseCard',
  }
)
