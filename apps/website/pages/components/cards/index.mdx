export function BlogIndex() {
  const currentPageTitle = "Card Components";
  return getPagesUnderRoute('/components/cards').filter(page => page.meta?.title !== currentPageTitle && page.frontMatter?.title !== currentPageTitle).map(page => {
    return (
      <ShowcaseCard title={page.meta?.title || page.frontMatter?.title || page.name} href={page.route}>
        <Image src={page.frontMatter?.image} alt={page.frontMatter?.title} width={500} height={500} />
      </ShowcaseCard>
    )
  })
}

export const ShowcaseCard = Object.assign(
  Cards.Card.bind(),
  {
    defaultProps: {
      image: true,
      arrow: true,
    },
    displayName: 'ShowcaseCard',
  }
)

<Cards>
<BlogIndex />
</Cards>
