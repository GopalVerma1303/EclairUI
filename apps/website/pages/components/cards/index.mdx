import { getPagesUnderRoute } from 'nextra/context'
import Link from 'next/link'

# Card Components

<p className="opacity-80 text-lg">A reusable UI element that displays key user information, typically including a profile picture, name, and brief bio or contact details.</p>

export function BlogIndex() {
  const currentPageTitle = "Card Components";
  return getPagesUnderRoute('/components/cards').filter(page => page.meta?.title !== currentPageTitle && page.frontMatter?.title !== currentPageTitle).map(page => {
    return (
      <ShowcaseCard title={page.meta?.title || page.frontMatter?.title || page.name} href={page.route}>
        <p className="opacity-80 text-justify text-xs" style={{ margin: '1rem' }}>
          {page.frontMatter?.description}{' '}
        </p>
      </ShowcaseCard>
    )
  })
}

export const ShowcaseCard = Object.assign(
  Cards.Card.bind(),
  {
    defaultProps: {
      image: true,
      arrow: true,
    },
    displayName: 'ShowcaseCard',
  }
)

<Cards>
<BlogIndex />
</Cards>
