import { getPagesUnderRoute } from 'nextra/context'
import Link from 'next/link'
import {Cards, Card} from "nextra/components"
import Image from 'next/image'

# Backgrounds

<p className="component-description">Beautifully designed backgrounds. Copy, paste, and customize to enhance your web projects effortlessly.</p>

<br />

export function BlogIndex() {
  const currentPageTitle = "Backgrounds";
  return getPagesUnderRoute('/components/backgrounds')
    .filter(page => page.meta?.title !== currentPageTitle && page.frontMatter?.title !== currentPageTitle)
    .map(page => {
      const fullUrl = page.frontMatter?.image;
      const srcPath = fullUrl?.replace(/^https?:\/\/[^/]+/, '');
      return (
        <ShowcaseCard title={page.meta?.title || page.frontMatter?.title || page.name} href={page.route}>
          <Image src={srcPath} alt={page.frontMatter?.title} width={500} height={300} />
        </ShowcaseCard>
      );
    });
}

export const ShowcaseCard = Object.assign(
  Cards.Card.bind(),
  {
    defaultProps: {
      image: true,
      arrow: true,
    },
    displayName: 'ShowcaseCard',
  }
)

<Cards>
<BlogIndex />
</Cards>
